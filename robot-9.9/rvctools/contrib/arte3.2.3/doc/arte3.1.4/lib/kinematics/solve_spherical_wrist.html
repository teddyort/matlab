solve_spherical_wrist

PURPOSE

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

SYNOPSIS

function q = solve_spherical_wrist(robot, q, T, wrist, method)

DESCRIPTION 

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   q = solve_spherical_wrist(robot, q, T, wrist)    
   Solves the inverse kinematic problem for a spherical wrist
   robot: robot structure.
   q: vector containing the values of the joints 1, 2 and 3.
   T: orientation of the last reference system.
   wrist: select -1 or 1 for two possible solutions (wrist up, wrist down)
   
    See also DIRECTKINEMATIC.

0029 function q = solve_spherical_wrist(robot, q, T, wrist, method)
0030 
0031 switch method
0032     
0033      %algebraic solution
0034     case 'algebraic'
0035         T01=dh(robot, q, 1);
0036         T12=dh(robot, q, 2);
0037         T23=dh(robot, q, 3);
0038         
0039         Q=inv(T23)*inv(T12)*inv(T01)*T;
0040         
0041         %detect the degenerate case when q(5)=0, this leads to zeros
0042         % in Q13, Q23, Q31 and Q32 and Q33=1
0043         
0044         %detect if q(5)==0
0045         if abs(Q(3,3)-1)>0.00001 
0046             %normal solution
0047             if wrist==1 %wrist up
0048                 q(4)=atan2(-Q(2,3),-Q(1,3));        
0049                 q(6)=atan2(-Q(3,2),Q(3,1));            
0050                 q(5)=atan2(-Q(3,2)/sin(q(6)),Q(3,3));
0051             else %wrist down
0052                 q(4)=atan2(-Q(2,3),-Q(1,3))-pi;            
0053                 q(6)=atan2(-Q(3,2),Q(3,1))+pi;            
0054                 q(5)=atan2(-Q(3,2)/sin(q(6)),Q(3,3));
0055             end
0056         else %degenerate solution, in this case, q4 cannot be determined,
0057              % so q(4)=0 is assigned
0058             if wrist==1 %wrist up
0059                 q(4)=0;
0060                 q(5)=0;
0061             else %wrist down
0062                 q(4)=-pi;
0063                 q(5)=0;
0064             end             
0065             q(6)=atan2(-Q(1,2)+Q(2,1),Q(1,1)+Q(2,2));
0066         end  
0067  
0068        %geometric solution
0069     case 'geometric' 
0070         % T is the noa matrix defining the position/orientation of the end
0071         % effector's reference system
0072         vx6=T(1:3,1);
0073         vz5=T(1:3,3); % The vector a z6=T(1:3,3) is coincident with z5
0074         
0075         % Obtain the position and orientation of the system 3
0076         % using the already computed joints q1, q2 and q3
0077         T01=dh(robot, q, 1);
0078         T12=dh(robot, q, 2);
0079         T23=dh(robot, q, 3);
0080         T03=T01*T12*T23;
0081          
0082         vx3=T03(1:3,1);
0083         vy3=T03(1:3,2);
0084         vz3=T03(1:3,3);
0085         
0086         % find z4 normal to the plane formed by z3 and a
0087         vz4=cross(vz3, vz5);    % end effector's vector a: T(1:3,3)
0088         
0089         % in case of degenerate solution,
0090         % when vz3 and vz6 are parallel--> then z4=0 0 0, choose q(4)=0 as solution
0091         if norm(vz4) <= 0.00001
0092             if wrist == 1 %wrist up
0093                 q(4)=0;
0094             else
0095                 q(4)=-pi; %wrist down
0096             end
0097         else
0098             %this is the normal and most frequent solution
0099             cosq4=wrist*dot(-vy3,vz4);
0100             sinq4=wrist*dot(vx3,vz4);
0101             q(4)=atan2(sinq4, cosq4);
0102         end
0103         %propagate the value of q(4) to compute the system 4
0104         T34=dh(robot, q, 4);
0105         T04=T03*T34;
0106         vx4=T04(1:3,1);
0107         vy4=T04(1:3,2);
0108              
0109         % solve for q5
0110         cosq5=dot(vy4,vz5);
0111         sinq5=dot(-vx4,vz5);
0112         q(5)=atan2(sinq5, cosq5);
0113         
0114         %propagate now q(5) to compute T05
0115         T45=dh(robot, q, 5);
0116         T05=T04*T45;
0117         vx5=T05(1:3,1);
0118         vy5=T05(1:3,2);
0119         
0120         % solve for q6
0121         cosq6=dot(vx6,vx5);
0122         sinq6=dot(vx6,vy5);
0123         q(6)=atan2(sinq6, cosq6);     
0124         
0125     
0126         
0127     otherwise
0128         disp('no method specified');
0129 end
